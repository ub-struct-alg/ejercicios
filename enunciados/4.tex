O(n,\mathbb{Z}) &:= \{M \in GL(n,\mathbb{Z}): M^TM=Id\}, \\
SO(n,\mathbb{Z}) &:= \{M \in O(n,\mathbb{Z}): det(M)=1\}, \\
\end{align*}

el grupo lineal, el grupo especial lineal, el grupo ortogonal y 
el grupo especial ortogonal respectivamente.

\end{parts}
