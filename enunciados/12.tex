\question
Sean $p,q$ numeros primos distintos y $r,s \in \mathbb{N}^*$:
\begin{parts}
\part Determina $\#\{x\in\mathbb{Z}/p\mathbb{Z}: \mathbb{Z}/p\mathbb{Z}=\langle x \rangle\}$.
\part Determina $\#\{x\in\mathbb{Z}/p^r\mathbb{Z}: \mathbb{Z}/p^r\mathbb{Z}=\langle x \rangle\}$.
\part Determina $\#\{x\in\mathbb{Z}/p^rq^s\mathbb{Z}: \mathbb{Z}/p^rq^s\mathbb{Z}=\langle x \rangle\}$.
\end{parts}
