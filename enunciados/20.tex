\question 
(El grupo de los cuaterniones) Sea $H_8$ el subgrupo de $GL(2,\mathbb{C})$ generado por las matrices 
\begin{align*}
Id:=
 \begin{pmatrix}
  1 & 0 \\
  0 & 1
 \end{pmatrix},
I:=
 \begin{pmatrix}
  i & 0 \\
  0 & -i
 \end{pmatrix},
J:=
 \begin{pmatrix}
  0 & 1 \\
  -1 & 0
 \end{pmatrix},
K:=
 \begin{pmatrix}
  0 & i \\
  i & 0
 \end{pmatrix}.
\end{align*}
\begin{parts}
\part Demuestra que $H_8$ es un grupo tal que $Id$ es el elemento neutro, $I^4=Id$, $I^2=J^2$ y $IJ=JI^3$.
\part Calcula el orden de cada elemento de $H_8$.
\part Demuestra $\langle a,b|a^4=e,a^2=b^2,ab=ba^3\rangle \simeq H_8$.
\end{parts}
