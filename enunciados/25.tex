\question 
Sean $f: G_1 \to G_2$ un morfismo de grupos, $H_1 \subseteq G_1$ y 
$H_2 \subseteq G_2$ subgrupos.
\begin{parts}
\part Demuestra que $f(H_1)$ es subgrupo de $G_2$ y $f^{-1}(H_2)$ de $G_1$.
\part Demuestra $H_1 \triangleleft G_2 \implies f^{-1}(H_2) \triangleleft G_1$.
\part Demuestra $H_2 \triangleleft G_1 \implies f(H_1) \triangleleft f(G_1)$;
pero no necesariamente $f(H_1)\triangleleft G_2$.
\end{parts}
